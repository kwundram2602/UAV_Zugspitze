\textbf{Whitebox predictor indices:}
\begin{itemize}
    \item Relative Topography Position (RTP) $\rightarrow$ (i.e. how elevated or low-lying a site is relative to its surroundings) \cite{newman-2018}.
    \item Windward Index (WI) $\rightarrow$ ( exposure of the land-surface to a hypothetical wind flux) \cite{Bhner2009LandSurfacePS}
    \item Directional Relief (Dir.\ Relief) $\rightarrow$ (elevation difference between a cell and its surroundings in a given compass direction) \cite{lapen-1993}
\end{itemize}
